   2. ENTITY CLASS (Student.java)
package com.example.demo;

import jakarta.persistence.*;

@Entity
@Table(name = "students")
class Student {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private int id;

    private String name;
    private int age;
    private String department;

    public Student() {}

    public int getId() { return id; }
    public void setId(int id) { this.id = id; }

    public String getName() { return name; }
    public void setName(String name) { this.name = name; }

    public int getAge() { return age; }
    public void setAge(int age) { this.age = age; }

    public String getDepartment() { return department; }
    public void setDepartment(String department) { this.department = department; }
}
   3. REPOSITORY (JpaRepository + Custom Query Methods)
package com.example.demo;

import java.util.List;
import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.jpa.repository.Query;

interface StudentRepository extends JpaRepository<Student, Integer> {

    // 1. Find by department
    List<Student> findByDepartment(String department);

    // 2. Find students whose age is greater than given age
    List<Student> findByAgeGreaterThan(int age);

    // 3. Find by department AND age
    List<Student> findByDepartmentAndAge(String department, int age);

    // 4. Custom Query using @Query
    @Query("SELECT s FROM Student s WHERE s.age < :age")
    List<Student> getStudentsYoungerThan(int age);
}
   4. SERVICE CLASS
package com.example.demo;

import java.util.List;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

@Service
class StudentService {

    @Autowired
    private StudentRepository repo;

    public List<Student> getByDepartment(String dept) {
        return repo.findByDepartment(dept);
    }

    public List<Student> getByAgeGreater(int age) {
        return repo.findByAgeGreaterThan(age);
    }

    public List<Student> getByDeptAndAge(String dept, int age) {
        return repo.findByDepartmentAndAge(dept, age);
    }

    public List<Student> getYoungerStudents(int age) {
        return repo.getStudentsYoungerThan(age);
    }
}

   5. CONTROLLER
package com.example.demo;

import java.util.List;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;

@RestController
@RequestMapping("/students")
class StudentController {

    @Autowired
    private StudentService service;

    // 1. Get by department
    @GetMapping("/department/{dept}")
    public List<Student> getByDepartment(@PathVariable String dept) {
        return service.getByDepartment(dept);
    }

    // 2. Get students age > given value
    @GetMapping("/age/{age}")
    public List<Student> getByAge(@PathVariable int age) {
        return service.getByAgeGreater(age);
    }

    // 3. Get by department AND age
    @GetMapping("/dept-age/{dept}/{age}")
    public List<Student> getByDeptAndAge(@PathVariable String dept, @PathVariable int age) {
        return service.getByDeptAndAge(dept, age);
    }

    // 4. Custom query (age less than)
    @GetMapping("/younger/{age}")
    public List<Student> getYounger(@PathVariable int age) {
        return service.getYoungerStudents(age);
    }
}
